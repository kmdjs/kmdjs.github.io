require(["SCJ", "canvas/structure"], function (SCJ) {
    var graphics = SCJ.createGraphics(450, 370, "#canvasCTT");  
    with (graphics) {
        frameRate(60);
        var yoff = 0,img = loadImage("../img/bg.png");
        tick(function () {
            image(img, 0, 0)  
            fps(0, 0);
            var prevXY = [], path = [], xoff = 0;      
            for (var x = 103; x <= 290; x += 5) {
                var y = map(noise(xoff, yoff), 0, 1, 190, 260);
                prevXY[0] = x;
                prevXY[1] = y;
                path.push([prevXY[0], prevXY[1]]);
                xoff += 0.05;            
            }
            yoff += 0.01;         
            path.push([(width / 2) - 100, height - 77, "bezier"]);
            path.push([path[0][0], path[0][1]]);
            fill(39, 255, 255, 0.8);
            polygon(path);
            stroke(255, 255, 255, 0.8);
            strokeWeight(2);
            for (var i = 0, j = path.length - 3; i < j; i++) {
                var item = path[i];
                var next = path[i + 1];
                line(item[0], item[1], next[0], next[1]);
            }
        })
    }
})

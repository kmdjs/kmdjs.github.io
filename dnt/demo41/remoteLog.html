require(["SCJ", "core/remoteLog"], function (SCJ) {
    var obj = {
        name: "zhanglei",
        age: 25,
        test: {
            test:
                { test: 2 },
            testArr:[1,2,3]
        }
    };
    SCJ.createRemoteLog({
        //给这次日志取个昵称，方便查询
        nickName: "hehe",
        //需要记录的对象，支持数字、文本、函数、对象等任何类型
        log: obj
    });
    //-----or-----
    //SCJ.cr({
    //    nickName: "hehe",
    //    log: obj
    //});
})

使用方式

require(["SCJ", "core/class"], function (SCJ) {
    var Person = SCJ.createClass({
        init: function (age, name) {
            this.age = age;
            this.name = name;
        },
        say: function () {
            console.log("my name is " + this.name);
        }
    });

    var Man = Person.extend({
        init: function (age, name, girlFriendCount) {
            this._super(age, name);
            this.girlFriendCount = girlFriendCount || 0;
        },
        chasingGirls: function () {
            this.girlFriendCount++;
            console.log(this.girlFriendCount)
        }
    })

    var p = new Person(25, "当耐特1");
    var m = new Man(25, "当耐特2", 1);
    p.say();
    //output:my name is 当耐特1 
    m.say();
    // output:my name is 当耐特2
    m.chasingGirls();
    //output:2 
})

使用方式

require(["SCJ", "core/dom"], function (SCJ, $) {
    //查找 ID 为"myDiv"的元素
    $("#myDiv");
    //查找一个 DIV 元素
    $("div");
    //查找所有类是 "myClass" 的元素
    $(".myClass");
    //找到每一个元素
    $("*")
    //找到匹配任意一个类的元素
    $("div,span,p.myClass")
    //找到表单中所有的 input 元素
    $("form input")
    //...
    //...
})

其他使用方式

require(["SCJ", "core/dom"], function (SCJ) {
    //查找 ID 为"myDiv"的元素
    SCJ.createQuery("#myDiv");
    //---------------or---------------
    //查找一个 DIV 元素
    SCJ.cq("div");
    //---------------or---------------
    //查找所有类是 "myClass" 的元素
    var $ = SCJ.createQuery;
    $(".myClass");
    //..
    //...
    //...
})

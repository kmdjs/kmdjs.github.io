使用方式

require(["SCJ", "core/animate"], function (SCJ, $) {
    SCJ.createAnimate({
        //运动的元素
        target: SCJ.createQuery("div"),
        //动画的结果
        to: {
            left: 100,
            top: 200
        },
        //执行时间
        duration: 1000,
        //速率变化
        easing: "easeInOutBounce",
        //结束回调
        onAnimateEnd: function () {
            alert("运动结束回调")
        }
    })
})

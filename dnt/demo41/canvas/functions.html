使用方式

require(["SCJ", "canvas/structure"], function (SCJ) {
    var graphics = SCJ.createGraphics(200, 200, "#canvasCTT");
    graphics.background(151);
    graphics.noLoop();

    graphics.tick(function () {
        draw_target(68, 34, 200, 10);
        draw_target(152, 16, 100, 3);
        draw_target(100, 144, 80, 5);
    })

    function draw_target(xloc, yloc, size, num) {
        var grayvalues = 255 / num;
        var steps = size / num;
        for (var i = 0; i < num; i++) {
            graphics.fill(i * grayvalues);
            graphics.ellipse(xloc, yloc, size - i * steps, size - i * steps);
        }
    }
})

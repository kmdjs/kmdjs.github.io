使用方式

require(["SCJ", "canvas/structure"], function (SCJ) {
    var graphics = SCJ.createGraphics(640, 360, "#canvasCTT");
    var subGraphics = SCJ.createGraphics(400, 200);
    graphics.image(subGraphics, 120, 60);
   
    graphics.tick(function () {        
        graphics.fill(0, 255);
        graphics.rect(0, 0, graphics.width, graphics.height);
        graphics.fill(255);
        graphics.circle(graphics.mouseX, graphics.mouseY, 30);
   
        subGraphics.background(51);
        subGraphics.stroke(255);
        subGraphics.circle(graphics.mouseX - 120, graphics.mouseY - 60, 30);
    })
})

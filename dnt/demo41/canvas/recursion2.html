使用方式

require(["SCJ", "canvas/structure"], function (SCJ) {
    var graphics = SCJ.createGraphics(200, 200, "#canvasCTT");
    graphics.background(204);
    graphics.noLoop();

    drawCircle(100, 100, 80, 8);

    function drawCircle( x,  y,  radius,  level) {                    
        var tt = 126 * level/6.0;
        graphics.fill(tt, 153);
        graphics.ellipse(x, y, radius*2, radius*2);      
        if(level > 1) {
            level = level - 1;
            var num = graphics.randomInt(2, 6);
            for(var i=0; i<num; i++) {
                var a = graphics.random(0, graphics.TWO_PI);
                var nx = x + graphics.cos(a) * 6.0 * level;
                var ny = y + graphics.sin(a) * 6.0 * level;
                drawCircle(nx, ny, radius/2, level);
            }
        }
    }
})

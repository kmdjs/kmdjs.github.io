require(["SCJ", "canvas/structure"], function (SCJ) {
    var graphics = SCJ.createGraphics(200, 200, "#canvasCTT");
    graphics.background(0);
    
    graphics.stroke(255);
    graphics.point(graphics.width / 2, graphics.height / 2);
    graphics.point(graphics.width / 2, graphics.height / 4);
    graphics.point(60, 30);
    graphics.point(60, 134);
    graphics.point(160, 50);
    graphics.point(280, -800);
    graphics.point(201, 100);

    graphics.stroke(204);
    graphics.line(0, 73, graphics.width, 73);

    graphics.rect(110, 55, 40, 36);
})

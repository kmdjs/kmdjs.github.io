使用方式

require(["SCJ", "canvas/structure"], function (SCJ) {
    var graphics = SCJ.createGraphics(200, 200, "#canvasCTT");
    graphics.background(204);
    graphics.noLoop();

    drawCircle(126, 170, 6);

    function drawCircle(x, radius, level) {
        var tt = 126 * level / 4.0;
        graphics.fill(tt);
        graphics.ellipse(x, 100, radius * 2, radius * 2);
        if (level > 1) {
            level = level - 1;
            drawCircle(x - radius / 2, radius / 2, level);
            drawCircle(x + radius / 2, radius / 2, level);
        }
    }
})

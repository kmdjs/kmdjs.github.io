使用方式

require(["SCJ", "canvas/structure"], function (SCJ) {
    var graphics = SCJ.createGraphics(200, 200, "#canvasCTT");
    graphics.background(127);
    
    for (var i = 0; i < graphics.height; i += 20) {
        graphics.fill(0);
        graphics.rect(0, i, graphics.width, 10);
        graphics.fill(255);
        graphics.rect(i, 0, 10, graphics.height);
    }
})

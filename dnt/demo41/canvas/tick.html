使用方式

require(["SCJ", "canvas/structure"], function (SCJ) {
    var graphics = SCJ.createGraphics(200, 200, "#canvasCTT");
    graphics.stroke(255);     
    graphics.frameRate(60);

    var y = 100;
    graphics.tick(function () {
        graphics.background(0);   
        y = y - 1;
        if (y < 0) { y = graphics.height; }
        graphics.line(0, y, graphics.width, y);
    })
})

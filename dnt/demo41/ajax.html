使用方式

require(["SCJ", "core/ajax"], function (SCJ) {
    SCJ.createAJAX({
        type: "POST",
        url: "some.php",
        data: "name=John&location=Boston",
        success: function (msg) {
            alert("Data Saved: " + msg);
        }
    });
    //-----------or-----------
    SCJ.ca({
        type: "POST",
        url: "some.php",
        data: "name=John&location=Boston",
        success: function (msg) {
            alert("Data Saved: " + msg);
        }
    });
})
